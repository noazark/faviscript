var context = this var w = context.width var h = context.height var centerX = w / 2 var centerY = h / 2 var pad = Math.round(w * 0.10) function arc(scale, color) { context.beginPath() context.arc(centerX, centerY, (w-pad*2)/2, 0, scale * Math.PI, false); context.fillStyle = color context.fill() context.closePath() } arc(4/3, '#59f')
